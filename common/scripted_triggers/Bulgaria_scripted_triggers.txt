BUL_has_zveno_rule = {
	custom_trigger_tooltip = {
		tooltip = BUL_has_zveno_rule_tooltip
		OR = {
			AND = {
				NOT = { has_country_flag = BUL_based_path }
				has_government = national_populist
			}
			AND = {
				has_country_flag = georgiev_zveno
				has_government = totalist
			}
			AND = {
				has_government = paternal_autocrat
				has_country_flag = BUL_zveno_is_pataut
			}
			AND = {
				has_government = social_democrat
				has_country_flag = kazasov_zveno
			}
			AND = {
				has_government = radical_socialist
				has_country_flag = kazasov_zveno
			}
		}
		NOT = {
			has_country_flag = BUL_IMRO_coup
			has_country_flag = BUL_tsankov_victory
		}
	}
}

BUL_has_socialist_rule = {
	custom_trigger_tooltip = {
		tooltip = BUL_syndicalist_government
		has_socialist_government = yes
		BUL_has_zveno_rule = no
		NOT = {
			has_country_flag = BUL_based_path
			has_country_flag = BUL_IMRO_coup
			has_country_flag = BUL_banu_is_tot
			has_country_flag = BUL_banu_is_radsoc
		}
	}
}
